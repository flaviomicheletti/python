```python
def verbing(s):
  if "ing" in s:
    return s + "ly"
  elif len(s) >= 3:
    return s + "ing"
  else:
    return s

# (google solution)
def verbingV2(s):
  if len(s) >= 3:
    if s[-3:] != 'ing': s = s + 'ing'
    else: s = s + 'ly'
  return s

def not_bad(s):
  n = s.find('not')
  b = s.find('bad')
  if n != -1 and b != -1 and b > n:
    s = s[:n] + 'good' + s[b+3:]
  return s

def front_back(a, b):
  a_middle = int(len(a) / 2)
  b_middle = int(len(b) / 2)

  # adiciona 1 se o tamanho for ímpar
  if len(a) % 2 == 1:
    a_middle = a_middle + 1

  if len(b) % 2 == 1:
    b_middle = b_middle + 1

  return a[:a_middle] + b[:b_middle] + a[a_middle:] + b[b_middle:]
```
